// تقرير فواتير الكاشير /api/cashier/report-invoices
app.MapGet("/api/cashier/report-invoices", async (AppDbContext db, HttpRequest request) =>
{
    var query = db.CashierInvoices
        .AsNoTracking()
        .AsQueryable();

    var dateFromStr   = request.Query["dateFrom"].ToString();
    var dateToStr     = request.Query["dateTo"].ToString();
    var paymentMethod = request.Query["paymentMethod"].ToString();

    if (DateTime.TryParse(dateFromStr, out var dateFrom))
    {
        dateFrom = dateFrom.Date;
        query = query.Where(i => i.InvoiceDate >= dateFrom);
    }

    if (DateTime.TryParse(dateToStr, out var dateTo))
    {
        dateTo = dateTo.Date.AddDays(1).AddTicks(-1);
        query = query.Where(i => i.InvoiceDate <= dateTo);
    }

    if (!string.IsNullOrWhiteSpace(paymentMethod) && paymentMethod != "all")
    {
        query = query.Where(i => i.PaymentMethod == paymentMethod);
    }

    var data = await query
        .OrderByDescending(i => i.Id)
        .Select(i => new
        {
            id            = i.Id,
            invoiceDate   = i.InvoiceDate,
            paymentMethod = i.PaymentMethod,
            customerName  = i.CustomerName,
            customerPhone = i.CustomerPhone,
            subTotal      = i.SubTotal,
            discountTotal = i.DiscountTotal,
            vatTotal      = i.VatTotal,
            grandTotal    = i.GrandTotal,
            isSuspended   = i.IsSuspended,
            notes         = i.Notes
        })
        .ToListAsync();

    return Results.Ok(data);
});

// تقرير خصومات الفواتير /api/reports/invoice-discounts
app.MapGet("/api/reports/invoice-discounts", async (AppDbContext db) =>
{
    try
    {
        var invoices = await db.CashierInvoices
            .AsNoTracking()
            .ToListAsync();

        var summary = new
        {
            invoicesCount = invoices.Count,
            totalSub      = invoices.Sum(x => (double)x.SubTotal),
            totalDiscount = invoices.Sum(x => (double)x.DiscountTotal),
            totalVat      = invoices.Sum(x => (double)x.VatTotal),
            totalGrand    = invoices.Sum(x => (double)x.GrandTotal)
        };

        return Results.Ok(summary);
    }
    catch (Exception ex)
    {
        Console.WriteLine("Error in /api/reports/invoice-discounts: " + ex);

        var emptySummary = new
        {
            invoicesCount = 0,
            totalSub      = 0d,
            totalDiscount = 0d,
            totalVat      = 0d,
            totalGrand    = 0d,
            error         = ex.Message
        };

        // نرجّع 200 عشان المتصفح ما يطلع رسالة فشل، لكن القيم كلها صفر
        return Results.Ok(emptySummary);
    }
});

#region EF Core Models

public class AppDbContext : DbContext
{
    public DbSet<Product>            Products            => Set<Product>();
    public DbSet<Customer>           Customers           => Set<Customer>();
    public DbSet<CashierInvoice>     CashierInvoices     => Set<CashierInvoice>();
    public DbSet<CashierInvoiceItem> CashierInvoiceItems => Set<CashierInvoiceItem>();

    public AppDbContext(DbContextOptions<AppDbContext> options) : base(options) { }

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        modelBuilder.Entity<Product>().ToTable("Products");
        modelBuilder.Entity<Customer>().ToTable("Customers");
        modelBuilder.Entity<CashierInvoice>().ToTable("CashierInvoices");
        modelBuilder.Entity<CashierInvoiceItem>().ToTable("CashierInvoiceItems");

        modelBuilder.Entity<CashierInvoice>()
            .HasMany(i => i.Items)
            .WithOne(i => i.Invoice!)
            .HasForeignKey(i => i.InvoiceId)
            .OnDelete(DeleteBehavior.Cascade);
    }
}
